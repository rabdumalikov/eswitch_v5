749  {
750  constexpr auto generic_lambda = []< typename T, T... ints >
751  ( std::index_sequence< ints... > && int_seq, auto && tup, auto && f )
752  {
753  ( f( std::get< ints >( tup ) ), ... );
754  };

762 
763  generic_lambda(
764  std::make_index_sequence< sizeof...( Cnds ) >{},
765  details::move_default_case_to_the_end( std::forward< Cnds >( cnds )... ),

768  using namespace details;
769 
770  if( fallthrough && !( *fallthrough ) ) return;

772  constexpr auto amount_args = amount_args_v< decltype( cnd.func ) >;
773 
774  if constexpr( amount_args == 0 )
775  {
776  if( fallthrough && *fallthrough )
777  {
778  cnd.func();
779 
780  fallthrough = cnd.fallthrough;
781 
782  return;
783  }
784  }
785 
786  auto res = cnd.cnd( tup_ );
787 
788  if( !res ) return;
789 
790  if constexpr( amount_args == 1 && !std::is_same_v< decltype( res ), bool > )
791  {
792  if constexpr( has_return_value )
793  {
794  return_value = static_cast< return_t >( cnd.func( std::move( *res ) ) );
795  }
796  else
797  {
798  cnd.func( std::move( *res ) );
799  }
800  }
801  else if constexpr( amount_args == 0 )
802  {
803  if constexpr( has_return_value )
804  {
805  return_value = static_cast< return_t >( cnd.func() );
806  }
807  else
808  {
809  cnd.func();
810  }
811  }
812  else
813  {
814  unreachable();
815  }
816 
817  fallthrough = cnd.fallthrough;
818  });
819 
820  if constexpr( has_return_value )
821  {
822  return return_value.value();
823  }
824  }
