436  struct Polymorphism_support
437  {
438  template< typename TupleEntry >
439  inline static constexpr auto execute( TupleEntry && tuple_entry, const CaseEntry & ) noexcept
440  requires has_type< CaseEntry >
441  {
442  using _T = std::remove_reference_t< TupleEntry >;
443 
444  using type = std::conditional_t< std::is_const_v< std::remove_pointer_t< _T > >,
445  const typename CaseEntry::type, typename CaseEntry::type >;

451  return d != nullptr ? std::make_optional( d ) : std::optional< type* >{};
452  }
453  else

786  {

798  std::make_index_sequence< sizeof...( Cnds ) >{},
799  details::move_default_case_to_the_end( std::forward< Cnds >( cnds )... ),

804  if( fallthrough && !( *fallthrough ) ) return;

806  constexpr auto amount_args = amount_args_v< decltype( cnd.func ) >;
807 
808  if constexpr( amount_args == 0 )

810  if( fallthrough && *fallthrough )
811  {
812  cnd.func();
813 
814  fallthrough = cnd.fallthrough;
815 
816  return;
817  }
818  }
819 
820  auto res = cnd.cnd( tup_ );
821 
822  if( !res ) return;
823 
824  if constexpr( amount_args == 1 && !std::is_same_v< decltype( res ), bool > )
825  {
826  if constexpr( has_return_value )
827  {
828  return_value = static_cast< return_t >( cnd.func( std::move( *res ) ) );
829  }
830  else
831  {
832  cnd.func( std::move( *res ) );
833  }
834  }
835  else if constexpr( amount_args == 0 )
836  {
837  if constexpr( has_return_value )
838  {
839  return_value = static_cast< return_t >( cnd.func() );
840  }
841  else
842  {
843  cnd.func();
844  }
845  }
846  else
847  {
848  unreachable();
849  }
850 
851  fallthrough = cnd.fallthrough;
852  });
